# head variable

Returns all but the last element of the given list or string.

Signature:

head: {
    (val: string): string;
    <T extends unknown[]>(val: T): T;
}

# Example 1

// String
head('hello') // 'hell'
head('h') // ''
head('') // ''

# Example 2

head([1, 2, 3]) // [1, 2]
head(['hello', 'world']) // ['hello']
head(['hello']) // []
head([]) // []
