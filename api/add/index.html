# add variable

Adds first argument and second argument.

Signature:

add: {
    (a: number, b: number): number;
    (a: number): (b: number) => number;
    (a: bigint, b: bigint): bigint;
    (a: bigint): (b: bigint) => bigint;
}

# Example 1

// Basic
add(1, 2) // 3

# Example 2

// Bigint
add(1n, 2n) // 3n

# Example 3

// Curry
const plus2(2)
plus2(-3) // -1
