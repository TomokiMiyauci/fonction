# multiply variable

Multiplies first argument and second argument.

Signature:

multiply: {
    (a: number, b: number): number;
    (a: number): (b: number) => number;
    (a: bigint, b: bigint): bigint;
    (a: bigint): (b: bigint) => bigint;
}

# Example 1

// Basic
multiply(2, 3) // 6

# Example 2

// Bigint
multiply(2n, 3n) // 6n

# Example 3

// Curry
const double = multiply(2)
double(4) // 8
