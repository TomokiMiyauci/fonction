# tail variable

Returns all but the first element of the given list or string.

Signature:

tail: {
    (val: string): string;
    <T extends unknown[]>(val: T): T;
}

# Example 1

// String
tail('hello') // 'ello'
tail('h') // ''
tail('') // ''

# Example 2

tail([1, 2, 3]) // [2, 3]
tail(['hello', 'world']) // ['world']
tail(['hello']) // []
tail([]) // []
