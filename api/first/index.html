# first variable

Returns the first element of the given list or string.

Signature:

first: <T extends string | unknown[]>(val: T) => First<T>

# Example 1

// String
first('hello') // 'h'

# Example 2

// Array
first('hello', 'new', 'world') // 'hello'
first([]) // undefined
first(['one', 2, 3, 4]) // 'one'
